const connection = require('./../database/dbConnection');
 
const getSkills = (Obj, cb) => {
  const sql = {
    text: `SELECT * FROM skills where user_id =$1`,
//     values: [Obj.user_id]
    values: [Obj]
 
  };
  connection.query(sql, (err, res) => {
    if (err) {
      cb(err);
    } else {
      cb(null, res.rows);
    }
  });
};

const getSkill = (skill, cb) => {
  const sql = {
    text: `SELECT skill FROM skills WHERE skill = $1`,
    values: [skill]
  };
  connection.query(sql, (err, res) => {
    if (err) {
      cb(err);
    } else {
      cb(null, res.rows[0]);
    }
  });
};
 
const checkSkill = (skill, cb) => {
  getSkill(skill, (err, res) => {
    if (err) {
      cb(err);
    } else {
      if (res) {
        cb(null, res.skill === skill);
      } else {
        cb(null, false);
      }
    }
  });
};
 
const deleteSkill = (Obj, cb) => {

      cb(null, res);
    }
  });
};
